function suma(a, b){
    return a+b;
}

function resta(a, b){
    return a-b;
}

function division(a, b){
    if(b<>0)
        return a/b;
    else
        print("Error");
}

function multiplicar(a, b){
    return a*b;
}

function potencia(a, b){
    return a ^ b;
}

function raiz(a,b){
    if(b<>0)
        return Math.Pow(a, 1/b);
    else
        print("Error");
    
}
